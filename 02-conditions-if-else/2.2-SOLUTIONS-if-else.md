# LES CONDITIONS IF / ELSE

## Exercice 1

```js
const temperature = parseFloat(prompt("Entrez la température actuelle :"));

if (temperature < 14) {
    console.log("Froid");
} else if (temperature >= 14 && temperature <= 25) {
    console.log("Modéré");
} else if (temperature > 25) {
    console.log("Chaud");
} else {
    console.log("Température non valide.");
}
```

## Exercice 4

```js
let poids = prompt("Entrez votre poids en kg :").replace(",", ".");
let taille = prompt("Entrez votre taille en mètres :").replace(",", ".");

// Convertir les saisies en nombres flottants
poids = parseFloat(poids);
taille = parseFloat(taille);

if (isNaN(poids) || isNaN(taille)) {
    console.log("Veuillez entrer des valeurs numériques valides.");
} else {
    const imc = poids / (taille * taille);
    const imcArrondi = imc.toFixed(2);

    console.log(`Votre IMC est de ${imcArrondi}`);

    if (imc >= 18.5 && imc <= 24.9) {
        console.log("IMC normal");
    } else if (imc >= 25.0 && imc <= 29.9) {
        console.log("Surpoids");
    } else if (imc >= 30.0 && imc <= 34.9) {
        console.log("Obésité modérée");
    } else if (imc >= 35.0 && imc <= 39.9) {
        console.log("Obésité sévère");
    } else if (imc >= 40.0) {
        console.log("Obésité massive");
    } else {
        console.log("IMC inférieur à la normale.");
    }
}
```
